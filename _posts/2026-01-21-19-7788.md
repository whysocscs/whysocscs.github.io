<pre class="cpp" data-ke-language="cpp" data-ke-type="codeblock" id="code_1768925491216"><code>#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

char names[1000001][6];

int compare(const void *a, const void *b) {
    return strcmp((char *)b, (char *)a);
}

int main() {
    int n;
    
    if (scanf("%d", &amp;n) != 1) return 0;
    
    for (int i = 0; i &lt; n; i++) {
        scanf("%s %*s", names[i]);
    }

    qsort(names, n, sizeof(names[0]), compare);

    for (int i = 0; i &lt; n; ) {
        if (i &lt; n - 1 &amp;&amp; strcmp(names[i], names[i+1]) == 0) {
            i += 2; 
        } 
        else {
            printf("%s\n", names[i]);
            i++; 
        }
    }

    return 0;
}</code></pre>
